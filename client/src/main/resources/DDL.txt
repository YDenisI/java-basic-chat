CREATE SCHEMA TESTS


CREATE TABLE tests (
    id SERIAL PRIMARY KEY,             -- Уникальный идентификатор теста
    title VARCHAR(255) NOT NULL,      -- Название теста
    description TEXT                  -- Описание теста (опционально)

)

CREATE TABLE questions (
    id SERIAL PRIMARY KEY,                     					-- Уникальный идентификатор вопроса
    test_id SMALLINT REFERENCES tests(id) ON DELETE CASCADE, 	-- Идентификатор теста
    question_text TEXT NOT NULL                					-- Текст вопроса
)

CREATE TABLE answers (
    id SERIAL PRIMARY KEY,                     					-- Уникальный идентификатор ответа
    question_id INT REFERENCES questions(id) ON DELETE CASCADE, -- Идентификатор вопроса
    answer_text TEXT NOT NULL,                 					-- Текст ответа
    is_correct BOOLEAN DEFAULT FALSE            				-- Флаг правильного ответа (по умолчанию - ложный)
)
